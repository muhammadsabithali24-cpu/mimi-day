<script>

function start() {

    document.getElementById("intro").style.display = "none";

    startCountdown();

}


// Countdown

let num = 5;

function startCountdown() {

    let cd = document.getElementById("countdown");

    cd.style.display = "flex";

    cd.innerHTML = num;


    let interval = setInterval(() => {

        num--;

        cd.innerHTML = num;


        if (num === 0) {

            clearInterval(interval);

            cd.style.display = "none";

            showWish();

        }

    }, 1000);

}


// Show Birthday Wish

function showWish() {

    let wish = document.getElementById("bigWish");

    wish.style.display = "flex";

    createConfetti();


    setTimeout(() => {

        wish.style.display = "none";

        showMain();

    }, 2500);

}


// Confetti effect

function createConfetti() {

    for (let i = 0; i < 80; i++) {

        let conf = document.createElement("div");

        conf.className = "confetti";

        conf.style.left = Math.random() * 100 + "vw";

        conf.style.background = ["#ff4d88","#ffd700","#00c3ff","#7cffcb"][Math.floor(Math.random()*4)];

        document.body.appendChild(conf);

    }

}


// Main message

let mainText = "You are the best thing that happened in my life 🤍\n\nThank you for being with me… always.\n\nI truly pray that all your wishes come true on this beautiful day ✨\n\nI just want to keep celebrating your birthdays with you, again and again 💕";

let finalText = "I pray all your dreams and wishes come true. With love, yours Mu 💝";


let i = 0;

function typeMessage() {

    if (i < mainText.length) {

        document.getElementById("typewriter").innerHTML += mainText.charAt(i);

        i++;

        setTimeout(typeMessage, 50);

    } else {

        setTimeout(showFinalPop, 500);

    }

}


// Final pop with orbiting sparkles

function showFinalPop() {

    let pop = document.createElement("div");

    pop.style.position = "fixed";

    pop.style.top = "0";

    pop.style.left = "0";

    pop.style.width = "100%";

    pop.style.height = "100%";

    pop.style.background = "radial-gradient(circle, #fff0f5, #ffd6e0)";

    pop.style.display = "flex";

    pop.style.justifyContent = "center";

    pop.style.alignItems = "center";

    pop.style.flexDirection = "column";

    pop.style.zIndex = "9999";


    let textEl = document.createElement("div");

    textEl.className = "finalPop";

    textEl.innerHTML = finalText;

    pop.appendChild(textEl);

    document.body.appendChild(pop);


    createConfetti();

    floating();


    // Add orbiting sparkles

    for (let j=0;j<20;j++){

        let sparkle = document.createElement("div");

        sparkle.className = "orbit";

        sparkle.innerHTML = ["✨","💖","🌸","💫"][Math.floor(Math.random()*4)];

        sparkle.style.top = "50%";

        sparkle.style.left = "50%";

        let radius = 50 + Math.random()*50; // radius in px

        sparkle.style.setProperty('--radius', radius + 'px');

        sparkle.style.animationDuration = (2 + Math.random()*2) + "s";

        pop.appendChild(sparkle);

    }


    setTimeout(() => { pop.remove(); }, 6000);

}


// Floating hearts animation

function floating() {

    setInterval(() => {

        let el = document.createElement("div");

        el.className = "heart";

        el.innerHTML = ["💖","🦋","✨","🌸"][Math.floor(Math.random()*4)];

        el.style.left = Math.random()*100 + "vw";

        el.style.fontSize = "25px";

        el.style.animationDuration = (5 + Math.random()*5) + "s";

        document.body.appendChild(el);

    }, 400);

}


function showMain() {

    document.getElementById("main").style.display = "block";

    typeMessage();

    floating();

}
